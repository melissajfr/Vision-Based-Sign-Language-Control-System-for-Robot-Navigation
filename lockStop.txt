// ========================================
// ULTRASONIC SENSOR FUNCTIONS
// ========================================

float quickDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
   
  long duration = pulseIn(echoPin, HIGH, 5000);
  if (duration == 0) return 999;
  return duration * 0.034 / 2;
}

void sendSensorData() {

  Blynk.virtualWrite(V2, rightDistance);

  Blynk.virtualWrite(V6, (rightDistance > 0 && rightDistance < THRESHOLD) ? "⚠️ OBS" : "Clear");
  Blynk.virtualWrite(V7, (backDistance > 0 && backDistance < THRESHOLD) ? "⚠️ OBS" : "Clear");
  Blynk.virtualWrite(V8, (leftDistance > 0 && leftDistance < THRESHOLD) ? "⚠️ OBS" : "Clear");
}

void stopMotors() {
  ledcWrite(enA, 0);
  ledcWrite(enB, 0);
  digitalWrite(in1, LOW);
  digitalWrite(in2, LOW);
  digitalWrite(in3, LOW);
  digitalWrite(in4, LOW);
  currentCommand = "STOP";
  Serial.println("🛑 STOP");
}

void moveForward() {
  Serial.print("   [Checking FRONT: ");
  Serial.print(frontDistance);
  Serial.println(" cm]");
  
  if (frontDistance <= THRESHOLD && frontDistance > 0) {
    Serial.print("🚫 FRONT BLOCKED (");
    Serial.print(frontDistance);
    Serial.println(" cm) - LOCKED!");
    stopMotors();
    blockedDirection = "FORWARD";
    return;
  }
  
  if (blockedDirection == "FORWARD") {
    Serial.println("🔒 Still blocked! Change direction first!");
    stopMotors();
    return;
  }
  
  digitalWrite(in1, HIGH);
  digitalWrite(in2, LOW);
  digitalWrite(in3, HIGH);
  digitalWrite(in4, LOW);
  ledcWrite(enA, SPEED);
  ledcWrite(enB, SPEED);
  currentCommand = "FORWARD";
  blockedDirection = "NONE";

void processCommand(char cmd) {
  lastCommandTime = millis();
  
  Serial.print("📥 Command: ");
  Serial.print(cmd);
  Serial.print(" → ");
  
  switch(cmd) {
    case 'F': 
      moveForward(); 
      break;
    case 'B': 
      moveBackward(); 
      break;
    case 'L': 
      turnLeft(); 
      break;
    case 'R': 
      turnRight(); 
      break;
    case 'S': 
      stopMotors();
      blockedDirection = "NONE";
      break;
    default: 
      Serial.println("❌ Unknown"); 
      break;
  }
}

void setup() {
  Serial.begin(115200);
  delay(100);
   
  Serial.println("\n========================================");
  Serial.println("🔒 LOCKED OBSTACLE MODE + DEBUG");
  Serial.println("========================================");

  // Motor pins
  pinMode(in1, OUTPUT);
  pinMode(in2, OUTPUT);
  pinMode(in3, OUTPUT);
  pinMode(in4, OUTPUT);
  ledcAttachChannel(enA, freq, resolution, pwmChannelA);
  ledcAttachChannel(enB, freq, resolution, pwmChannelB);
  stopMotors();

  // Sensor pins
  pinMode(TRIG1, OUTPUT);
  pinMode(ECHO1, INPUT);
  pinMode(TRIG2, OUTPUT);
  pinMode(ECHO2, INPUT);
  pinMode(TRIG3, OUTPUT);
  pinMode(ECHO3, INPUT);
  pinMode(TRIG4, OUTPUT);
  pinMode(ECHO4, INPUT);

  Serial.println("✅ Hardware initialized");
  Serial.println();
  Serial.println("📍 Sensor Pin Assignment:");
  Serial.println("   FRONT: TRIG=19, ECHO=21");
  Serial.println("   RIGHT: TRIG=16, ECHO=5");
  Serial.println("   BACK:  TRIG=18, ECHO=4  ← CHECK THIS!");
  Serial.println("   LEFT:  TRIG=15, ECHO=2");
  Serial.println();

  // WiFi & Blynk
  Serial.print("📡 Connecting WiFi...");
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);
   
  Serial.println("\n✅ WiFi Connected!");
  Serial.println("✅ Blynk Connected!");
  Serial.print("📍 IP: ");
  Serial.println(WiFi.localIP());
  Serial.println("========================================");
  Serial.print("   ESP32_IP = '");
  Serial.print(WiFi.localIP());
  Serial.println("'");
  Serial.println("========================================");

  server.begin();
  timer.setInterval(1000L, sendSensorData);

  Serial.println("✅ All systems ready!");
  Serial.println("========================================");
  Serial.println("\n🔍 DEBUG MODE ACTIVE");
  Serial.println("Sensor readings will print every 2 seconds");
  Serial.println("========================================\n");
}

// ========================================
// MAIN LOOP
// ========================================

void loop() {
  Blynk.run();
  timer.run();
   
  // Read all sensors
  frontDistance = quickDistance(TRIG1, ECHO1);
  rightDistance = quickDistance(TRIG2, ECHO2);
  backDistance = quickDistance(TRIG3, ECHO3);  // ← BACK sensor
  leftDistance = quickDistance(TRIG4, ECHO4);
  
  // ========================================
  // DEBUG: Print sensor values every 2 seconds
  // ========================================
  if (millis() - lastDebugTime > 2000) {
    Serial.println("\n━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    Serial.println("🔍 SENSOR READINGS:");
    Serial.println("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    
    Serial.print("   FRONT: ");
    Serial.print(frontDistance);
    Serial.print(" cm");
    if (frontDistance <= THRESHOLD && frontDistance > 0) Serial.print("  ⚠️ OBSTACLE!");
    Serial.println();
    
    Serial.print("   RIGHT: ");
    Serial.print(rightDistance);
    Serial.print(" cm");
    if (rightDistance <= THRESHOLD && rightDistance > 0) Serial.print("  ⚠️ OBSTACLE!");
    Serial.println();
    
    Serial.print("   BACK:  ");
    Serial.print(backDistance);
    Serial.print(" cm");
    if (backDistance <= THRESHOLD && backDistance > 0) Serial.print("  ⚠️ OBSTACLE!");
    Serial.println();
    
    Serial.print("   LEFT:  ");
    Serial.print(leftDistance);
    Serial.print(" cm");
    if (leftDistance <= THRESHOLD && leftDistance > 0) Serial.print("  ⚠️ OBSTACLE!");
    Serial.println();
    
    Serial.println("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━");
    Serial.print("   Current: ");
    Serial.print(currentCommand);
    if (blockedDirection != "NONE") {
      Serial.print("  🔒 LOCKED: ");
      Serial.print(blockedDirection);
    }
    Serial.println();
    Serial.println("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n");
    
    lastDebugTime = millis();
  }
   
  // Handle Python commands
  if (!client || !client.connected()) {
    client = server.available();
  }
   
  if (client && client.available()) {
    char cmd = client.read();
    processCommand(cmd);
    client.print("OK\n");
  }

  // Continuous obstacle monitoring
  continuousObstacleCheck();

  // Safety timeout
  if (millis() - lastCommandTime > 2000 && currentCommand != "STOP") {
    Serial.println("⏱️  Timeout → Stop");
    stopMotors();
    blockedDirection = "NONE";
  }

  delay(10);
}
